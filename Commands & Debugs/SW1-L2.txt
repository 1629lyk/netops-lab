Stage 5 — SW1-L2
SW1-L2 console. This is a pure Layer 2 switch — no routing needed. Management goes via Vlan1, default gateway is SW2-L3's management IP.
enable
configure terminal

hostname SW1-L2

! No routing on access layer switch
no ip routing

! Uplink to SW2-L3 — trunk carrying VLANs 10 and 20
interface Ethernet0/0
 description TRUNK-TO-SW2-L3
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 1,10,20
 no shutdown

! Access port for Alpine-1 — VLAN 10
interface Ethernet0/1
 description ACCESS-ALPINE1
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
 no shutdown

! Access port for Alpine-2 — VLAN 20
interface Ethernet0/2
 description ACCESS-ALPINE2
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
 no shutdown

! VLAN definitions
vlan 10
 name CORPORATE
vlan 20
 name GUEST

! Use VLAN10 as management SVI
interface Vlan10
 ip address 10.10.0.254/24
 no shutdown

! Default gateway — SW2-L3 VLAN10 SVI
ip default-gateway 10.10.0.1

! SSH
ip domain-name netops-lab.local
crypto key generate rsa modulus 2048
username netops privilege 15 secret NetOps2024!
line vty 0 4
 login local
 transport input ssh
ip ssh version 2

end
write memory
