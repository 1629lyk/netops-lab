Stage 4 — SW2-L3
SW2-L3 console. Interface naming is Ethernet0/x (IOU style).
enable
configure terminal

hostname SW2-L3

! Enable routing — this is a Layer 3 switch
ip routing

! Uplink to ASA inside
interface Ethernet0/0
 description UPLINK-TO-ASA-INSIDE
 no switchport
 ip address 10.0.14.2 255.255.255.0
 no shutdown

! Downlink to SW1-L2 — trunk
interface Ethernet0/1
 description TRUNK-TO-SW1-L2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 no shutdown

! VLAN definitions
vlan 10
 name CORPORATE
vlan 20
 name GUEST

! Inter-VLAN SVIs
interface Vlan10
 description CORPORATE-USERS
 ip address 10.10.0.1 255.255.255.0
 no shutdown

interface Vlan20
 description GUEST-USERS
 ip address 10.20.0.1 255.255.255.0
 no shutdown

! Management SVI
interface Vlan1
 ip address 192.168.66.112 255.255.255.0
 no shutdown

! Default route toward ASA
ip route 0.0.0.0 0.0.0.0 10.0.14.1

! OSPF — participate in area 0
router ospf 1
 router-id 10.0.14.2
 network 10.0.14.0 0.0.0.255 area 0
 network 10.10.0.0 0.0.0.255 area 0
 network 10.20.0.0 0.0.0.255 area 0
 default-information originate

! SSH
ip domain-name netops-lab.local
crypto key generate rsa modulus 2048
username netops privilege 15 secret NetOps2024!
line vty 0 4
 login local
 transport input ssh
ip ssh version 2

end
write memory
